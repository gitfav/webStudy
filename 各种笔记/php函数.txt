CREATE TABLE `h_article_label` (
	`label_id` int(8) UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键id',
	`label_fid` int(10) NOT NULL COMMENT '标签的父类标签（0为没有）',
	`label_name` varchar(18) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '标签名',
	`label_icon` varchar(128) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '标签的图标',
	`label_jointime` int(11) NOT NULL COMMENT '字段加入时间',
	`label_uptime` int(11) NOT NULL COMMENT '字段更新时间',
	`label_status` tinyint(1) NOT NULL COMMENT '标签状态（-1为不显示）',
	PRIMARY KEY (`label_id`),
	UNIQUE KEY `label_id`(`label_id`) USING BTREE
) ENGINE=InnoDB
DEFAULT CHARACTER SET=utf8 COLLATE=utf8_general_ci
COMMENT='文章标签记录表';
